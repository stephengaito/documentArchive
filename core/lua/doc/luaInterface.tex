static int lua_core_lua_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  if (aKey) {
    getGitVersionInto(gitVersionKeyValues, aKey, aValue);
    lua_pushstring(lstate, aValue);
  } else {
    lua_pushstring(lstate, "no valid key provided");
  }
  return 1;
}

static int lua_core_lua_setVerbose(lua_State *lstate) {
  Boolean verbose = 
    (lua_toboolean(lstate, 1) ? TRUE : FALSE);
  getJoyLoLInterpInto(lstate, jInterp);
  jInterp->rootCtx->verbose = verbose;
  lua_pop(lstate, 1);
  return 0;
}

static int lua_core_lua_pushLoadPath(lua_State *lstate) {
  Symbol *aPath = lua_tostring(lstate, 1);
  getJoyLoLInterpInto(lstate, jInterp);
  pushLoadPath(jInterp, aPath);
  lua_pop(lstate, 1);
  return 0;
}

static int lua_core_lua_loadFile(lua_State *lstate) {
  Symbol *aFile = lua_tostring(lstate, 1);
  getJoyLoLInterpInto(lstate, jInterp);
  loadAFile(jInterp->rootCtx, aFile);
  lua_pop(lstate, 1);
  return 0;
}

static int lua_core_lua_runREPL(lua_State *lstate) {
  getJoyLoLInterpInto(lstate, jInterp);
  runREPLInContext(jInterp->rootCtx);
  return 0;
}

static const struct luaL_Reg lua_core_lua [] = {
  { "gitVersion",   lua_core_lua_getGitVersion },
  { "setVerbose",   lua_core_lua_setVerbose    },
  { "pushLoadPath", lua_core_lua_pushLoadPath  },
  { "loadFile",     lua_core_lua_loadFile      },
  { "runREPL",      lua_core_lua_runREPL       },
  {NULL, NULL}
};

static void coreLuaWriteStdOut(Symbol *aMessage) {
  fprintf(stdout, "%s\n", aMessage);
}

static void coreLuaWriteStdErr(Symbol *aMessage) {
  fprintf(stderr, "%s\n", aMessage);
}

static void *coreLuaCallback(size_t resourceId) {
  if (resourceId == JoyLoLCallback_StdOutMethod) {
    return (void*)coreLuaWriteStdOut;
  } else if (resourceId == JoyLoLCallback_StdErrMethod) {
    return (void*)coreLuaWriteStdErr;
  }
  return NULL;
} 

int luaopen_joylol_core_lua (lua_State *lstate) {
  stackDump(lstate);
  Symbol *verbose = getenv("JOYLOL_VERBOSE");
  if (verbose) printf("loading [joylol.jInterps]\n");
  stackDump(lstate);
  setJoyLoLCallbackFrom(lstate, coreLuaCallback);
  stackDump(lstate);
  lua_getglobal(lstate, "require");
  lua_pushstring(lstate, "joylol.jInterps");
  if (lua_pcall(lstate, 1, 1, 0)) {
    return luaL_error(lstate,
      "Failed to load [joylol.jInterps]\nERROR:\n%s\n",
      lua_tostring(lstate, -1)
    );
  }
  stackDump(lstate);
  if (verbose) printf("loaded [joylol.jInterps]\n");
  getJoyLoLInterpInto(lstate, jInterp);
  lua_pushstring(lstate, "core");
  lua_createtable(lstate,0, 1); // joylol.core 
  lua_pushstring(lstate, "lua");
  luaL_newlib(lstate, lua_core_lua);
  lua_settable(lstate, -3);
  lua_settable(lstate, -3);
  return 1;
}
