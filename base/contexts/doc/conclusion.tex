static int lua_contexts_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  const char* aValue = "no valid key provided";
  if (aKey) aValue = getGitVersion(gitVersionKeyValues, aKey);
  lua_pushstring(lstate, aValue);
  return 1;
}

static size_t joylol_register_contexts(lua_State *lstate) {
  CoAlgebra* theCoAlg    = (CoAlgebra*) calloc(1, sizeof(CoAlgebra));
  theCoAlg->name         = "Contexts";
  theCoAlg->registerFunc = joylol_register_contexts;
  theCoAlg->equalityFunc = NULL;
  theCoAlg->coAlgData    = NULL;
  registerCoAlgebra(lstate, theCoAlg);
  assert(theCoAlg->isA == ContextsTag);
  
  // in future... register Contexts words with JoyLoL dictionary. 
  
  return TRUE;
}

static const struct luaL_Reg lua_contexts [] = {
  {"gitVersion", lua_contexts_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_contexts (lua_State *lstate) {
  joylol_register_contexts(lstate);
  luaL_newlib(lstate, lua_contexts);
  return 1;
}
