\startCCode
Boolean extendJoyLoLImpl(
  JoyLoLInterp *jInterp,
  DictObj      *aDict,
  Symbol       *definedName,
  CFunction    *aFunc,
  Boolean       redefine
) {
  assert(aDict);

  DictNodeObj* aSym =
    createSymbolInThisDictionary(aDict, definedName);

  if (!aSym->value || redefine) {
    aSym->value =
      (JObj*)newCFunction(jInterp, aFunc);
  }

  return TRUE;
}

\startCCode
Boolean defineContextImpl(
  JoyLoLInterp *jInterp,
  DictObj      *aDict,
  Symbol       *definedName,
  ContextObj   *newCtx,
  Boolean       redefine
) {
  DEBUG(jInterp, "defineContext %p %p [%s] %p %zu\n",
        jInterp, aDict, definedName, newCtx, redefine);

  assert(aDict);
  
  DictNodeObj* aSym =
    createSymbolInThisDictionary(aDict, definedName);

  if (!aSym->value || redefine) {
    aSym->value   = (JObj*)newCtx;
  }

  return TRUE;
}
