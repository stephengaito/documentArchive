\section[title=JoyLoL words]

\startCCode
static void isCoAlgebraAP(ContextObj* aCtx) {
  assert(aCtx);
  JoyLoLInterp *jInterp = aCtx->jInterp;
  assert(jInterp);
  popCtxDataInto(aCtx, top);
  JObj* result = NULL;
  if (isCoAlgebra(top))
    result = newBoolean(jInterp, TRUE);
  else
    result = newBoolean(jInterp, FALSE);
  pushCtxData(aCtx, result);
}
\stopCCode

\setCHeaderStream{private}
\startCHeader
extern void registerCoAlgebraWords(JoyLoLInterp* jInterp);
\stopCHeader
\setCHeaderStream{public}

\startCCode
void registerCoAlgebraWords(JoyLoLInterp* jInterp) {
  assert(jInterp);
  extendJoyLoL(jInterp, "isCoAlgebra", isCoAlgebraAP);
}
\stopCCode
