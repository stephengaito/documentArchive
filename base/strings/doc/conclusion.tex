static int lua_strings_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  const char* aValue = "no valid key provided";
  if (aKey) aValue = getGitVersion(gitVersionKeyValues, aKey);
  lua_pushstring(lstate, aValue);
  return 1;
}

static size_t joylol_register_strings(lua_State *lstate) {
  CoAlgebra* theCoAlg    = (CoAlgebra*) calloc(1, sizeof(CoAlgebra));
  theCoAlg->name         = "Strings";
  theCoAlg->registerFunc = joylol_register_strings;
  theCoAlg->equalityFunc = NULL;
  theCoAlg->coAlgData    = NULL;
  registerCoAlgebra(lstate, theCoAlg);
  
  // in future... register Strings words with JoyLoL dictionary. 
  
  return TRUE;
}

static const struct luaL_Reg lua_strings [] = {
  {"gitVersion", lua_strings_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_strings (lua_State *lstate) {
  joylol_register_strings(lstate);
  luaL_newlib(lstate, lua_strings);
  return 1;
}
