\section[title=Lua interface]

\component gitVersion-c

\startCCode
static int lua_coAlgs_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  if (aKey) {
    getGitVersionInto(gitVersionKeyValues, aKey, aValue);
    lua_pushstring(lstate, aValue);
  } else {
    lua_pushstring(lstate, "no valid key provided");
  }
  return 1;
}

static const struct luaL_Reg lua_coAlgs [] = {
  {"gitVersion", lua_coAlgs_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_coAlgs (lua_State *lstate) {
  getJoyLoLInterpInto(lstate, jInterp);
  registerCoAlgs(jInterp);
  luaL_newlib(lstate, lua_coAlgs);
  return 1;
}
\stopCCode

In some instances, such as the typical \type{CTest} program 
\type{allCTests}, this Lua module (which can be \type{require}d as a 
shared library) is actually statically linked into the executable. In 
these cases we need the ability to mimic the standard Lua \type{require} 
process. The following \type{requireStaticallyLinkedCoAlgs} does just 
this. 

\startCHeader
Boolean requireStaticallyLinkedCoAlgs(
  lua_State *lstate
);
\stopCHeader

\startCCode
Boolean requireStaticallyLinkedCoAlgs(
  lua_State *lstate
) {
  lua_getglobal(lstate, "package");
  lua_getfield(lstate, -1, "loaded");
  luaopen_joylol_coAlgs(lstate);
  lua_setfield(lstate, -2, "joylol.coAlgs");
  lua_setfield(lstate, -2, "loaded");
  lua_pop(lstate, 1);
  return TRUE;
}
\stopCCode
