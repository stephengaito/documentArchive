static int lua_texts_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  const char* aValue = "no valid key provided";
  if (aKey) aValue = getGitVersion(gitVersionKeyValues, aKey);
  lua_pushstring(lstate, aValue);
  return 1;
}

static size_t joylol_register_texts(lua_State *lstate) {
  CoAlgebra* theCoAlg    = (CoAlgebra*) calloc(1, sizeof(CoAlgebra));
  theCoAlg->name         = "Texts";
  theCoAlg->registerFunc = joylol_register_texts;
  theCoAlg->equalityFunc = NULL;
  theCoAlg->coAlgData    = NULL;
  registerCoAlgebra(lstate, theCoAlg);
  assert(theCoAlg->isA == TextsTag);
  
  // in future... register Texts words with JoyLoL dictionary. 
  
  return TRUE;
}

static const struct luaL_Reg lua_texts [] = {
  {"gitVersion", lua_texts_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_texts (lua_State *lstate) {
  joylol_register_texts(lstate);
  luaL_newlib(lstate, lua_texts);
  return 1;
}
