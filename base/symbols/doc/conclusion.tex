static int lua_symbols_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  const char* aValue = "no valid key provided";
  if (aKey) aValue = getGitVersion(gitVersionKeyValues, aKey);
  lua_pushstring(lstate, aValue);
  return 1;
}

static size_t joylol_register_symbols(JoyLoLInterp* jInterp) {
  CoAlgebra* theCoAlg    = (CoAlgebra*) calloc(1, sizeof(CoAlgebra));
  theCoAlg->name         = "Symbols";
  theCoAlg->registerFunc = joylol_register_symbols;
  theCoAlg->equalityFunc = NULL;
  theCoAlg->printFunc    = NULL;
  theCoAlg->coAlgData    = NULL;
  registerCoAlgebra(jInterp, theCoAlg);

  registerSymbolWords(jInterp);

  return TRUE;
}

static const struct luaL_Reg lua_symbols [] = {
  {"gitVersion", lua_symbols_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_symbols (lua_State *lstate) {
  JoyLoLInterp* jInterp = getJoyLoLInterp(lstate);
  joylol_register_symbols(jInterp);
  luaL_newlib(lstate, lua_symbols);
  return 1;
}
