static int lua_booleans_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  const char* aValue = "no valid key provided";
  if (aKey) aValue = getGitVersion(gitVersionKeyValues, aKey);
  lua_pushstring(lstate, aValue);
  return 1;
}

static size_t joylol_register_booleans(lua_State *lstate) {
  registerCoAlgebra(
    lstate,
    "Booleans",
    joylol_register_booleans,
    NULL, // no equality function
    NULL  // no booleans-data
  );
  
  // in future... register Booleans words with JoyLoL dictionary. 
  
  return TRUE;
}

static const struct luaL_Reg lua_booleans [] = {
  {"gitVersion", lua_booleans_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_booleans (lua_State *lstate) {
  joylol_register_booleans(lstate);
  luaL_newlib(lstate, lua_booleans);
  return 1;
}
