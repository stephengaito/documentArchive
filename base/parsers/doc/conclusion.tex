static int lua_parsers_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  const char* aValue = "no valid key provided";
  if (aKey) aValue = getGitVersion(gitVersionKeyValues, aKey);
  lua_pushstring(lstate, aValue);
  return 1;
}

static size_t joylol_register_parsers(lua_State *lstate) {
  CoAlgebra* theCoAlg    = (CoAlgebra*) calloc(1, sizeof(CoAlgebra));
  theCoAlg->name         = "Parsers";
  theCoAlg->registerFunc = joylol_register_parsers;
  theCoAlg->equalityFunc = NULL;
  theCoAlg->coAlgData    = NULL;
  registerCoAlgebra(lstate, theCoAlg);
  assert(theCoAlg->isA == ParsersTag);
  
  // in future... register Parsers words with JoyLoL dictionary. 
  
  return TRUE;
}

static const struct luaL_Reg lua_parsers [] = {
  {"gitVersion", lua_parsers_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_parsers (lua_State *lstate) {
  joylol_register_parsers(lstate);
  luaL_newlib(lstate, lua_parsers);
  return 1;
}
