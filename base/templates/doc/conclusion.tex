static int lua_templates_getGitVersion (lua_State *lstate) {
  const char* aKey   = lua_tostring(lstate, 1);
  const char* aValue = "no valid key provided";
  if (aKey) aValue = getGitVersion(gitVersionKeyValues, aKey);
  lua_pushstring(lstate, aValue);
  return 1;
}

static size_t joylol_register_templates(lua_State *lstate) {
  CoAlgebra* theCoAlg    = (CoAlgebra*) calloc(1, sizeof(CoAlgebra));
  theCoAlg->name         = "Templates";
  theCoAlg->registerFunc = joylol_register_templates;
  theCoAlg->equalityFunc = NULL;
  theCoAlg->coAlgData    = NULL;
  registerCoAlgebra(lstate, theCoAlg);
  assert(theCoAlg->isA == TemplatesTag);
  
  // in future... register Templates words with JoyLoL dictionary. 
  
  return TRUE;
}

static const struct luaL_Reg lua_templates [] = {
  {"gitVersion", lua_templates_getGitVersion},
  {NULL, NULL}
};

int luaopen_joylol_templates (lua_State *lstate) {
  joylol_register_templates(lstate);
  luaL_newlib(lstate, lua_templates);
  return 1;
}
